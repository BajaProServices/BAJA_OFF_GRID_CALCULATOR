<script>
    // LISTA DE EQUIPOS PREDETERMINADOS
    const electrodomesticos = [
        { name: "Focos LED (Kit 10)", watts: 100 },
        { name: "Televisión LED", watts: 150 },
        { name: "Refrigerador Moderno", watts: 300 },
        { name: "Microondas", watts: 1200 },
        { name: "Licuadora", watts: 600 },
        { name: "Laptop / PC", watts: 200 },
        { name: "Ventilador", watts: 80 },
        { name: "Bomba de Pool (Alberca)", watts: 1500 }, // Añadido para pool
        { name: "Minisplit 1 Ton Inverter", watts: 1100 },
        { name: "Bomba de Agua (1/2 HP)", watts: 500 },
        { name: "Lavadora", watts: 600 },
        { name: "Cámaras de Seguridad (Kit)", watts: 80 }
    ];

    // DEFINICIÓN DE INVERSORES
    const inverters = [
        { label: "Inversor 3kW 48v", limit: 3000, price: 23330 },
        { label: "Inversor 6kW 48v", limit: 6000, price: 46600 },
        { label: "Inversor 12kW 48v", limit: 12000, price: 92000 }
    ];

    const batteryPrice = 32000; // 5kW Litio

    // INICIALIZACIÓN DE LA LISTA
    function init() {
        const container = document.getElementById('appliance-container');
        electrodomesticos.forEach((item, index) => {
            const div = document.createElement('div');
            div.className = 'appliance-item';
            div.onclick = () => toggleAppliance(index);
            div.innerHTML = `
                <input type="checkbox" id="item-${index}">
                <div class="appliance-info">
                    <span class="appliance-name">${item.name}</span>
                    <span class="appliance-watts">${item.watts} Watts</span>
                </div>
            `;
            container.appendChild(div);
        });
    }

    // SELECCIÓN DE EQUIPO
    function toggleAppliance(index) {
        const check = document.getElementById(`item-${index}`);
        check.checked = !check.checked;
        const parent = check.closest('.appliance-item');
        check.checked ? parent.classList.add('selected') : parent.classList.remove('selected');
        calculate();
    }

    // LÓGICA DE CÁLCULO
    function calculate() {
        let totalW = 0;
        electrodomesticos.forEach((item, index) => {
            if(document.getElementById(`item-${index}`).checked) totalW += item.watts;
        });

        document.getElementById('totalWatts').innerText = `${totalW.toLocaleString()} Wh`;

        // Cálculo Inversor
        let selectedInv = { label: "--", price: 0 };
        if(totalW > 0) {
            selectedInv = inverters.find(inv => totalW <= inv.limit) || inverters[2];
        }
        document.getElementById('recInverter').innerText = selectedInv.label;

        // Cálculo Baterías
        let numBatteries = totalW > 0 ? Math.ceil(totalW / 2500) : 0; 
        document.getElementById('recBatteries').innerText = `${numBatteries} Unidades`;

        // Cálculo Paneles
        let numPaneles = totalW > 0 ? Math.ceil((totalW * 1.3) / 550) : 0;
        document.getElementById('recPanels').innerText = `${numPaneles} Paneles`;

        // Total
        let totalBudget = totalW > 0 ? (selectedInv.price + (numBatteries * batteryPrice)) : 0;
        document.getElementById('totalBudget').innerText = `$${totalBudget.toLocaleString()}.00`;
    }

    // REINICIAR FORMULARIO
    function resetAll() {
        document.querySelectorAll('.appliance-item').forEach(el => el.classList.remove('selected'));
        document.querySelectorAll('input[type="checkbox"]').forEach(el => el.checked = false);
        calculate();
    }

    // ENVÍO DE DATOS POR WHATSAPP (A +5216122198325)
    function enviarCotizacion() {
        const nombre = document.getElementById('nombre').value;
        const tel = document.getElementById('telefono').value;
        const dir = document.getElementById('direccion').value;
        
        if(!nombre || !tel || !dir) {
            alert("Por favor completa Nombre, Teléfono y Dirección para solicitar tu cotización formal.");
            return;
        }

        const total = document.getElementById('totalWatts').innerText;
        const inv = document.getElementById('recInverter').innerText;
        const bat = document.getElementById('recBatteries').innerText;
        const pan = document.getElementById('recPanels').innerText;
        const money = document.getElementById('totalBudget').innerText;

        const msg = `*NUEVA SOLICITUD WEB - BAJA PROSERVICES*%0A` +
                    `---------------------------------%0A` +
                    `👤 *Cliente:* ${nombre}%0A` +
                    `📞 *Tel:* ${tel}%0A` +
                    `📍 *Ubicación:* ${dir}%0A` +
                    `---------------------------------%0A` +
                    `📊 *CARGA ESTIMADA:* ${total}%0A` +
                    `⚡ *INVERSOR:* ${inv}%0A` +
                    `🔋 *BATERÍAS:* ${bat}%0A` +
                    `☀️ *PANELES:* ${pan}%0A` +
                    `💰 *EQUIPO ESTIMADO:* ${money}%0A` +
                    `---------------------------------`;
        
        window.open(`https://wa.me/5216122198325?text=${msg}`, '_blank');
    }

    init();
</script>
